# Story 1.3: 错误处理和友好提示 - 实现计划

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** 为 Multiverse CLI 实现统一的错误处理机制，提供清晰的错误信息和解决建议

**Architecture:** 集中式错误处理器 + 启动前检测 + CLI 错误格式化输出。ErrorHandler 负责将错误码转换为友好信息，preflightChecks 在容器启动前快速失败，CLI 入口统一捕获并格式化输出。

```typescript
// packages/core/src/error-handler/error-handler.ts
import type { AppError, FormattedError } from './types.js';
import { ErrorCode } from './error-codes.js';
import { ERROR_TEMPLATES } from './error-templates.js';

export class ErrorHandler {
  format(error: AppError): FormattedError {
    const template = ERROR_TEMPLATES[error.code] || ERROR_TEMPLATES[ErrorCode.UNKNOWN_ERROR];
    
    return {
      title: template.title,
      description: this.replaceVariables(template.description, error.context),
      reason: template.reason,
      suggestions: template.suggestions.map(s => this.replaceVariables(s, error.context)),
      exitCode: template.exitCode,
    };
  }
  
  private replaceVariables(text: string, context?: Record<string, any>): string {
    if (!context) {
      return text;
    }
    
    let result = text;
    for (const [key, value] of Object.entries(context)) {
      result = result.replace(new RegExp(`\\{${key}\\}`, 'g'), String(value));
    }
    
    return result;
  }
}
```

```typescript
import { Given, When, Then } from '@cucumber/cucumber';
import { expect } from 'chai';
import { exec } from 'node:child_process';
import { promisify } from 'node:util';

const execAsync = promisify(exec);

Given('Docker is not running', async function() {
  // Mock or skip Docker
  this.mockDocker = true;
});

Given('Claude credentials do not exist', async function() {
  // Backup and remove credentials
  this.credentialsBackup = true;
});

Given('the workspace directory is read-only', async function() {
  // Set directory to read-only
  this.readOnlyWorkspace = true;
});

When('I run {string}', async function(command: string) {
  try {
    const result = await execAsync(command);
    this.output = result.stdout + result.stderr;
    this.exitCode = 0;
  } catch (error: any) {
    this.output = error.stdout + error.stderr;
    this.exitCode = error.code || 1;
  }
});

Then('the output should contain {string}', function(text: string) {
  expect(this.output).to.include(text);
});

Then('the exit code should be {int}', function(code: number) {
  expect(this.exitCode).to.equal(code);
});
```
