# Story 2.2 Template Drift Detection Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在不引入 watcher/daemon 的前提下，为 `start` 流程增加 template drift 检测和用户驱动同步，确保新 template 总是基于当前最新配置创建。

**Architecture:** 继续以 Template 作为不可变配置快照，在 template 中增加稳定 `fingerprint`。`core` 负责计算当前配置与 template 快照的指纹、生成 drift 摘要、创建派生 template、更新当前 verse 的 `templateId`；CLI 只负责在 `start` 中触发检测、提示用户并执行选择。E2E 先覆盖 `keep / sync-and-switch / cancel` 三条主路径，再补 core 与 CLI 单测。

**Tech Stack:** TypeScript, Vitest, Cucumber.js/Gherkin, Commander.js, Node.js `fs/promises`, Node.js `crypto`

```ts
// packages/core/src/template/template-fingerprint.ts
import { createHash } from 'node:crypto';
import type { ConfigSnapshot } from '@multiverse/types';

function normalizeText(content: string): string {
  return content.replace(/\r\n/g, '\n');
}

export function computeSnapshotFingerprint(snapshot: ConfigSnapshot): string {
  const lines: string[] = [];

  if (snapshot.claudeMd !== undefined) {
    lines.push('CLAUDE.md');
    lines.push(normalizeText(snapshot.claudeMd));
  }

  const files = [...snapshot.files].sort((a, b) => a.path.localeCompare(b.path));
  for (const file of files) {
    lines.push(`.claude/${file.path.replaceAll('\\', '/')}`);
    lines.push(normalizeText(file.content));
  }

  return createHash('sha256').update(lines.join('\n')).digest('hex');
}
```

```ts
// packages/core/src/template/config-drift.ts
import type { ConfigDriftCheckResult, ConfigSnapshot, Template } from '@multiverse/types';
import { createConfigSnapshot } from './config-snapshot.js';
import { computeSnapshotFingerprint } from './template-fingerprint.js';

function toMap(snapshot: ConfigSnapshot): Map<string, string> {
  const map = new Map<string, string>();
  if (snapshot.claudeMd !== undefined) {
    map.set('CLAUDE.md', snapshot.claudeMd);
  }
  for (const file of snapshot.files) {
    map.set(`.claude/${file.path}`, file.content);
  }
  return map;
}

export async function checkTemplateDrift({
  homeDir,
  template,
}: {
  homeDir: string;
  template: Template;
}): Promise<ConfigDriftCheckResult> {
  const currentSnapshot = await createConfigSnapshot(homeDir);
  const currentFingerprint = computeSnapshotFingerprint(currentSnapshot);
  const templateFingerprint = template.fingerprint || computeSnapshotFingerprint(template.snapshot);

  const currentMap = toMap(currentSnapshot);
  const templateMap = toMap(template.snapshot);
  const paths = [...new Set([...currentMap.keys(), ...templateMap.keys()])].sort();

  const addedFiles: string[] = [];
  const modifiedFiles: string[] = [];
  const removedFiles: string[] = [];

  for (const path of paths) {
    if (!templateMap.has(path)) {
      addedFiles.push(path);
    } else if (!currentMap.has(path)) {
      removedFiles.push(path);
    } else if (currentMap.get(path) !== templateMap.get(path)) {
      modifiedFiles.push(path);
    }
  }

  return {
    templateId: template.id,
    templateName: template.name,
    templateFingerprint,
    currentFingerprint,
    isDrifted: currentFingerprint !== templateFingerprint,
    addedFiles,
    modifiedFiles,
    removedFiles,
  };
}
```

```ts
// packages/cli/src/commands/start.ts
async function promptForDriftAction(summary: ConfigDriftCheckResult): Promise<'keep' | 'sync-and-switch' | 'cancel'> {
  process.stdout.write(
    `Current global config has drifted from template "${summary.templateName}".\n` +
      `Modified: ${summary.modifiedFiles.length}, Added: ${summary.addedFiles.length}, Removed: ${summary.removedFiles.length}\n`,
  );

  const examples = [...summary.modifiedFiles, ...summary.addedFiles, ...summary.removedFiles].slice(0, 5);
  if (examples.length > 0) {
    process.stdout.write(`Examples: ${examples.join(', ')}\n`);
  }

  process.stdout.write(
    '\nChoose an action:\n1. Keep using current template\n2. Create a new template from current config and switch this verse\n3. Cancel\n> ',
  );

  const answer = await readSingleLineFromStdin();
  if (answer === '2') return 'sync-and-switch';
  if (answer === '3') return 'cancel';
  return 'keep';
}
```
